\newtheorem{theorem}{Theorem}

\title{Two-sided ideals of $\mathcal{L}(V)$}
\date{}

\begin{document}
\maketitle

\begin{theorem}
Suppose $V$ is finite-dimensional. The only two-sided ideals of $\mathcal{L}(V)$ are $\{0\}$ and $\mathcal{L}(V)$.
\end{theorem}

\begin{proof}
Let $\mathcal{E}$ be a two-sided ideal of $\mathcal{L}(V)$. If $\mathcal{E} = \{0\}$ we are done,
so assume $\mathcal{E}$ is nonzero. We show $\mathcal{E} = \mathcal{L}(V)$.

\medskip
\noindent\textbf{Step 1: Setup.}
Since $\mathcal{E} \neq \{0\}$, there exists a nonzero $E \in \mathcal{E}$.
Since $E$ is not the zero map, there exists $u \in V$ with $Eu \neq 0$.
Extend $u$ to a basis $u, f_2, \ldots, f_n$ of $V$, and extend $Eu$ to a basis
$Eu, g_2, \ldots, g_n$ of $V$.

\medskip
\noindent\textbf{Step 2: Using the ideal to build controlled operators.}
Let $v_1, \ldots, v_n$ be a basis of $V$ and let $F \in \mathcal{L}(V)$ be arbitrary.
For each $i = 1, \ldots, n$, we construct an operator $A_i \in \mathcal{E}$ satisfying
\[
    A_i(v_j) = \begin{cases} Fv_i & \text{if } j = i, \\ 0 & \text{if } j \neq i. \end{cases}
\]
Define $R \in \mathcal{L}(V)$ by $R(v_i) = u$ and $R(v_j) = 0$ for $j \neq i$.
Define $S \in \mathcal{L}(V)$ by $S(Eu) = Fv_i$ and $S(g_k) = 0$ for $k \geq 2$.
Set $A_i = SER$.

The two-sided ideal property gives $A_i \in \mathcal{E}$ in two steps:
\begin{align*}
    E \in \mathcal{E} &\implies ER \in \mathcal{E} \quad \text{(closed under right-multiplication by } R\text{)}, \\
    ER \in \mathcal{E} &\implies S(ER) \in \mathcal{E} \quad \text{(closed under left-multiplication by } S\text{)}.
\end{align*}
We verify that $A_i = SER$ behaves as claimed on each basis vector $v_j$:
\[
    A_i(v_i) = S(E(R(v_i))) = S(E(u)) = S(Eu) = Fv_i,
\]
\[
    A_i(v_j) = S(E(R(v_j))) = S(E(0)) = 0 \quad \text{for all } j \neq i.
\]
The key idea is that $R$ routes the input $v_i$ to $u$ (the vector we know $E$
acts on nontrivially), then $E$ maps $u$ to $Eu$, and finally $S$ routes $Eu$
to any desired target $Fv_i$.

\medskip
\noindent\textbf{Step 3: Every operator in $\mathcal{L}(V)$ lies in $\mathcal{E}$.}
Applying Step 2 for each $i = 1, \ldots, n$ yields operators $A_1, \ldots, A_n \in \mathcal{E}$.
We claim $F = A_1 + \cdots + A_n$. Indeed, for each basis vector $v_i$,
\[
    (A_1 + \cdots + A_n)(v_i)
    = \sum_{k=1}^{n} A_k(v_i)
    = A_i(v_i)
    = Fv_i,
\]
where every term $A_k(v_i)$ with $k \neq i$ vanishes by construction. Since $F$ and
$A_1 + \cdots + A_n$ agree on every basis vector, they are equal as linear maps:
\[
    F = A_1 + \cdots + A_n.
\]
Since $\mathcal{E}$ is a subspace of $\mathcal{L}(V)$, it is closed under addition,
and each $A_i \in \mathcal{E}$, so $F \in \mathcal{E}$.

\medskip
Since $F \in \mathcal{L}(V)$ was arbitrary, every operator lies in $\mathcal{E}$,
so $\mathcal{L}(V) \subseteq \mathcal{E}$. Since also $\mathcal{E} \subseteq \mathcal{L}(V)$
by assumption, we conclude $\mathcal{E} = \mathcal{L}(V)$.
\end{proof}
